COPIE E COLE AS REGRAS ABAIXO NO SEU CONSOLE FIREBASE (Aba Rules):

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // 1. PERMISSÃO DE VERIFICAÇÃO (CRÍTICA):
    // Permite que qualquer pessoa leia a coleção 'leaders'.
    // Necessário para o app validar se o e-mail existe antes de criar a conta.
    match /leaders/{leader} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // 2. PROTEÇÃO DAS DEMAIS COLEÇÕES:
    // Exige login para Membros, PGs, Fotos, Escalas e Solicitações.
    match /members/{doc} { allow read, write: if request.auth != null; }
    match /pgs/{doc} { allow read, write: if request.auth != null; }
    match /pg_photos/{doc} { allow read, write: if request.auth != null; }
    match /change_requests/{doc} { allow read, write: if request.auth != null; }
    match /sectors/{doc} { allow read, write: if request.auth != null; }
    match /chaplains/{doc} { allow read, write: if request.auth != null; }
    match /collaborators/{doc} { allow read, write: if request.auth != null; }
    match /settings/{doc} { allow read, write: if request.auth != null; }
    match /meeting_schedules/{doc} { allow read, write: if request.auth != null; }

    // 3. ADMINISTRAÇÃO MASTER:
    // Permite acesso total ao e-mail do administrador.
    match /{document=**} {
      allow read, write: if request.auth != null && request.auth.token.email == "pastorescopel@gmail.com";
    }
  }
}
